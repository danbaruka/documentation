  <TabItem value="web" label="Web (CosmJS)">

```ts
import { StargateClient } from '@cosmjs/stargate';

const RPC = 'https://rpc.testnet.safrochain.com:443';
const DENOM = 'usaf';

const client = await StargateClient.connect(RPC);
console.log(await client.getHeight(), await client.getChainId());

const address = 'addr_safro1xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx';
console.log(await client.getBalance(address, DENOM));

const validators = await client.getValidators();
console.log(validators.slice(0, 3).map((v) => v.description?.moniker));
```

## Wasm smart query preview

```bash
REST="https://rest.testnet.safrochain.com"
curl -s "$REST/cosmwasm/wasm/v1/contract/addr_safro1contract.../smart/$(echo -n '{}' | base64)"
```
